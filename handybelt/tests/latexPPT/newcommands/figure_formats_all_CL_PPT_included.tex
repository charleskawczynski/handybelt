% Full FIGS (all captions/labels included):

\newcommand{\PPToneFig}[2]{
\FIG[!ht] \CE
	\IG[width=0.6\TW]{#1} \caption{#2}
\EFIG
}

\newcommand{\PPTtwoFigs}[4]{
\FIG[!ht]\CE
	\MP{0.48\TW} \CE
		\IG[width=.99\TW,height=0.3\TH,keepaspectratio]{#1} \caption{#2}
	\EMP
	\hspace{1em}
	\MP{0.48\TW} \CE
		\IG[width=.99\TW,height=0.3\TH,keepaspectratio]{#3} \caption{#4}
	\EMP
	\hspace{1em}
\EFIG
}
