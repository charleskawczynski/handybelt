% Full FIGS (all captions/labels included):

\newcommand{\oneFig}[3]{
\FIG[!ht] \CE
	\IG[width=0.6\TW]{#1} \caption{#2} \label{#3}
\EFIG
}

\newcommand{\twoFigs}[6]{
\FIG[!ht]\CE
	\MP{0.48\TW} \CE
		\IG[width=.99\TW,height=0.3\TH,keepaspectratio]{#1} \caption{#2} \label{#3}
	\EMP
	\hspace{1em}
	\MP{0.48\TW} \CE
		\IG[width=.99\TW,height=0.3\TH,keepaspectratio]{#4} \caption{#5} \label{#6}
	\EMP
	\hspace{1em}
\EFIG
}

\newcommand{\twoSubFigs}[8]{
\FIG[!ht]\CE
	\MP{0.48\TW}
		\SFIG[b]{\TW} \CE
			\IG[width=.99\TW,height=0.3\TH,keepaspectratio]{#1} \caption{#2} \label{#3}
		\ESFIG
	\EMP
	\hspace{.5em}
	\MP{0.48\TW}
		\SFIG[b]{\TW} \CE
			\IG[width=.99\TW,height=0.3\TH,keepaspectratio]{#4} \caption{#5} \label{#6}
		\ESFIG
	\EMP
	\vspace{1em}
	\caption{#7} \label{#8}
\EFIG
}
